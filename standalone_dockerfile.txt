# Dockerfile for Standalone PDF Translation Service
# Includes models and dependencies for offline operation

FROM python:3.11-slim

# Set working directory
WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    # For PDF processing
    libmupdf-dev \
    libfreetype6-dev \
    libjpeg-dev \
    libopenjp2-7-dev \
    # For fonts and text processing
    fonts-liberation \
    fonts-dejavu-core \
    # Utilities
    curl \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements and install Python dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy the translation scripts
COPY de_en-translator_v2.py ./pdf_translator.py
COPY offline_model_downloader.py .
COPY simple_test.py .

# Copy the pre-downloaded models (this is the key difference)
COPY models/ ./models/

# Create necessary directories
RUN mkdir -p /app/input /app/output /app/temp

# Copy Docker entrypoint
COPY docker-entrypoint.sh .
RUN chmod +x docker-entrypoint.sh

# Create non-root user for security
RUN groupadd -r pdfuser && useradd -r -g pdfuser pdfuser
RUN chown -R pdfuser:pdfuser /app
USER pdfuser

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD python simple_test.py

# Expose volume mounts for input/output
VOLUME ["/app/input", "/app/output"]

# Set environment variables
ENV MODEL_PATH="/app/models/Helsinki-NLP/opus-mt-de-en"
ENV PYTHONPATH="/app"
ENV OFFLINE_MODE="true"

# Default command
ENTRYPOINT ["./docker-entrypoint.sh"]
CMD ["--help"]
